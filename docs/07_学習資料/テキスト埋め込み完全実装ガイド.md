## 2. ステップ1: テキスト入力

### 2.1 基本的なテキスト正規化

```python
import unicodedata
import re

def normalize_text(text):
    """
    テキストの正規化処理
    - Unicode正規化 (NFC)
    - 小文字変換
    - 制御文字除去
    """
    # Unicode正規化
    text = unicodedata.normalize('NFC', text)
    
    # 小文字変換
    text = text.lower()
    
    # 制御文字を除去
    text = ''.join(char for char in text if unicodedata.category(char) != 'Cc')
    
    # 複数スペースを1つに統一
    text = re.sub(r'\s+', ' ', text)
    
    return text.strip()

# 使用例
sample_texts = [
    "  自然言語処理は　楽しい！  ",
    "ＮＬＰ処理（テスト）",
    "Hello   WORLD!!!"
]

for text in sample_texts:
    normalized = normalize_text(text)
    print(f"元: {repr(text)}")
    print(f"正規化後: {repr(normalized)}")
    print()

# 出力:
# 元: '  自然言語処理は　楽しい！  '
# 正規化後: '自然言語処理は楽しい！'
#
# 元: 'ＮＬＰ処理（テスト）'
# 正規化後: 'nlp処理（テスト）'
#
# 元: 'Hello   WORLD!!!'
# 正規化後: 'hello world!!!'
```

```python
import re

class SimpleRegexTokenizer:
    """正規表現を使った簡単なトークナイザー"""
    
    def tokenize(self, text):
        """
        単純な分割（単語と記号を区別）
        """
        # 単語（連続する英数字）と記号を分割
        pattern = r'\w+|[^\w\s]'
        tokens = re.findall(pattern, text)
        return tokens
```

```python
import MeCab

class MeCabTokenizer:
    """日本語用の形態素解析トークナイザー"""
    
    def __init__(self):
        self.mecab = MeCab.Tagger()
    
    def tokenize(self, text):
        """
        MeCab で形態素解析
        """
        tokens = []
        node = self.mecab.parseToNode(text)
        
        while node:
            # 形態素情報を取得
            surface = node.surface  # 表記
            features = node.feature.split(',')
            pos = features[0]  # 品詞
            base = features[6] if len(features) > 6 else surface  # 基本形
            
            # 名詞、動詞、形容詞等を扱う（記号は除外）
            if pos in ['名詞', '動詞', '形容詞', '副詞', '助詞']:
                tokens.append(base)
            
            node = node.next
        
        return tokens

# 使用例
try:
    tokenizer = MeCabTokenizer()
    text = "自然言語処理は難しい。でも学ぶ価値がある。"
    tokens = tokenizer.tokenize(text)
    print(f"入力: {text}")
    print(f"トークン: {tokens}")
    # 出力:
    # 入力: 自然言語処理は難しい。でも学ぶ価値がある。
    # トークン: ['自然', '言語', '処理', '難しい', '学ぶ', '価値']
except:
    print("MeCabがインストールされていません")
    print("インストール: pip install mecab-python3")
```

```python
from transformers import BertTokenizer

class BertWordPieceTokenizer:
    """BERT の WordPiece トークナイザー"""
    
    def __init__(self, model_name='bert-base-multilingual-cased'):
        self.tokenizer = BertTokenizer.from_pretrained(model_name)

        """トークン化"""
        tokens = self.tokenizer.tokenize(text)
        return tokens
    
    def get_token_ids(self, text):
        """トークンID化"""
        token_ids = self.tokenizer.encode(text)
        return token_ids
    
    def decode_ids(self, token_ids):
        """IDからテキストに戻す"""
        text = self.tokenizer.decode(token_ids)
        return text
```

## 4. ステップ3: トークンID変換

### 4.1 語彙（Vocabulary）の構築

```python
class Vocabulary:
    """単語から ID へのマッピング"""
    
    def __init__(self, min_freq=1):
        self.word2id = {}
        self.id2word = {}
        self.min_freq = min_freq
        self._add_special_tokens()
    
    def _add_special_tokens(self):
        """特殊トークンを追加"""
        special_tokens = ['<PAD>', '<UNK>', '<START>', '<END>']
        for token in special_tokens:
            idx = len(self.word2id)
            self.word2id[token] = idx
            self.id2word[idx] = token
    
    def build_from_texts(self, texts, max_vocab_size=10000):
        """
        テキストコーパスから語彙を構築
        
        Args:
            texts: トークンのリストのリスト
            max_vocab_size: 最大語彙サイズ
        """
        # 単語の出現頻度をカウント
        word_freq = {}
        for tokens in texts:
            for token in tokens:
                word_freq[token] = word_freq.get(token, 0) + 1
        
        # 頻度でフィルタ
        filtered_words = [
            w for w, f in word_freq.items() 
            if f >= self.min_freq
        ]
        
        # 頻度でソート（多い順）
        sorted_words = sorted(
            filtered_words, 
            key=lambda w: word_freq[w], 
            reverse=True
        )
        
        # 語彙に追加（最大サイズまで）
        for word in sorted_words[:max_vocab_size]:
            if word not in self.word2id:
                idx = len(self.word2id)
                self.word2id[word] = idx
                self.id2word[idx] = word
    
    def encode(self, tokens):
        """トークンをIDに変換"""
        ids = []
        for token in tokens:
            if token in self.word2id:
                ids.append(self.word2id[token])
            else:
                ids.append(self.word2id['<UNK>'])  # 未知語
        return ids
    
    def decode(self, ids):
        """IDをトークンに戻す"""
        tokens = []
        for idx in ids:
            if idx in self.id2word:
                tokens.append(self.id2word[idx])
            else:
                tokens.append('<UNK>')
        return tokens
    
    def __len__(self):
        return len(self.word2id)
    
    def save(self, filepath):
        """語彙を保存"""
        import json
        with open(filepath, 'w', encoding='utf-8') as f:
            json.dump(self.word2id, f, ensure_ascii=False, indent=2)
    
    def load(self, filepath):
        """語彙を読み込み"""
        import json
        with open(filepath, 'r', encoding='utf-8') as f:
            self.word2id = json.load(f)
            self.id2word = {v: k for k, v in self.word2id.items()}

# 使用例
vocab = Vocabulary(min_freq=1)

# サンプルテキストからトークンのリストを作成
sample_token_lists = [
    ['自然', '言語', '処理', 'は', '難しい'],
    ['機械', '学習', 'は', '面白い'],
    ['深層', '学習', 'は', '強力', 'だ'],
    ['自然', '言語', '処理', 'と', '機械', '学習', 'は', '関連', 'がある']
]

vocab.build_from_texts(sample_token_lists, max_vocab_size=100)

print("語彙サイズ:", len(vocab))
print("word2id:", vocab.word2id)
print()

# トークン → ID
test_tokens = ['自然', '言語', '処理', '未知語']
token_ids = vocab.encode(test_tokens)
print(f"トークン: {test_tokens}")
print(f"ID: {token_ids}")
print()

# ID → トークン
recovered_tokens = vocab.decode(token_ids)
print(f"復元トークン: {recovered_tokens}")

# 出力例:
# 語彙サイズ: 19
# word2id: {
#   '<PAD>': 0,
#   '<UNK>': 1,
#   '<START>': 2,
#   '<END>': 3,
#   '自然': 4,
#   '言語': 5,
#   ...
# }
#
# トークン: ['自然', '言語', '処理', '未知語']
# ID: [4, 5, 6, 1]  ← 未知語は<UNK>(1)に
#
# 復元トークン: ['自然', '言語', '処理', '<UNK>']
```

### 4.2 バッチ処理とパディング

```python
import numpy as np

class BatchProcessor:
    """バッチ処理とパディング"""
    
    def __init__(self, vocab, pad_token='<PAD>'):
        self.vocab = vocab
        self.pad_idx = vocab.word2id[pad_token]
    
    def encode_texts(self, texts, max_length=None):
        """
        複数テキストをエンコード＋パディング
        
        Args:
            texts: トークンのリストのリスト
            max_length: パディングの目標長（Noneで自動）
        
        Returns:
            numpy配列 (batch_size, max_length)
        """
        # トークンをIDに変換
        encoded = [self.vocab.encode(tokens) for tokens in texts]
        
        # 最大長を決定
        if max_length is None:
            max_length = max(len(ids) for ids in encoded)
        
        # パディング
        padded = np.full((len(encoded), max_length), self.pad_idx, dtype=np.int32)
        
        for i, ids in enumerate(encoded):
            length = min(len(ids), max_length)
            padded[i, :length] = ids[:length]
        
        return padded
    
    def create_attention_mask(self, padded_ids):
        """
        Attention マスク（実際のトークンは1、パディングは0）
        """
        mask = (padded_ids != self.pad_idx).astype(np.int32)
        return mask

# 使用例
processor = BatchProcessor(vocab, pad_token='<PAD>')

batch_texts = [
    ['自然', '言語', '処理'],
    ['機械', '学習', 'は', '面白い'],
    ['深層', '学習']
]

# エンコード
padded_ids = processor.encode_texts(batch_texts, max_length=5)
print("パディング後のID:")
print(padded_ids)
print("形状:", padded_ids.shape)
print()

# Attention マスク
attention_mask = processor.create_attention_mask(padded_ids)
print("Attention マスク:")
print(attention_mask)

# 出力例:
# パディング後のID:
# [[ 4  5  6  0  0]
#  [ 7  8  9 10  0]
#  [11 12  0  0  0]]
# 形状: (3, 5)
#
# Attention マスク:
# [[1 1 1 0 0]
#  [1 1 1 1 0]
#  [1 1 0 0 0]]
```

## 5. ステップ4: 埋め込みベクトル

### 5.1 埋め込み行列の初期化

```python
import numpy as np
import torch
import torch.nn as nn

class EmbeddingLayer:
    """埋め込み層の実装"""
    
    def __init__(self, vocab_size, embedding_dim=768, device='cpu'):
        """
        Args:
            vocab_size: 語彙サイズ
            embedding_dim: 埋め込み次元（BERT=768, GPT2=768など）
            device: CPU or GPU
        """
        self.embedding_layer = nn.Embedding(vocab_size, embedding_dim)
        self.device = device
        self.embedding_layer = self.embedding_layer.to(device)
    
    def forward(self, token_ids):
        """
        トークンIDから埋め込みベクトルを取得
        
        Args:
            token_ids: 形状 (batch_size, seq_length) のテンソル
        
        Returns:
            形状 (batch_size, seq_length, embedding_dim) のテンソル
        """
        if isinstance(token_ids, np.ndarray):
            token_ids = torch.from_numpy(token_ids).long().to(self.device)
        
        embeddings = self.embedding_layer(token_ids)
        return embeddings
    
    def get_embedding_matrix(self):
        """埋め込み行列全体を取得"""
        return self.embedding_layer.weight.detach().cpu().numpy()

# 使用例
vocab_size = len(vocab)
embedding_dim = 768

embedding_layer = EmbeddingLayer(vocab_size, embedding_dim, device='cpu')

# バッチID（例）
batch_ids = np.array([
    [4, 5, 6, 0, 0],     # シーケンス1
    [7, 8, 9, 10, 0],    # シーケンス2
    [11, 12, 0, 0, 0]    # シーケンス3
])

# 埋め込み取得
embeddings = embedding_layer.forward(batch_ids)

print("埋め込みの形状:", embeddings.shape)
print("(batch_size=3, seq_length=5, embedding_dim=768)")
print()

# 最初のシーケンスの最初のトークンの埋め込み
first_embedding = embeddings[0, 0, :]
print("最初のトークンの埋め込み（最初の10次元）:")
print(first_embedding[:10].detach().cpu().numpy())
print()

# 埋め込み行列全体の形状
embedding_matrix = embedding_layer.get_embedding_matrix()
print("埋め込み行列の形状:", embedding_matrix.shape)
print(f"({vocab_size}語彙 × {embedding_dim}次元)")

# 出力例:
```

### 5.2 埋め込みの性質

```python
def analyze_embeddings(embeddings, token_ids, vocab):
    """埋め込みの類似度分析"""
    from scipy.spatial.distance import cosine
    
    # テンソルをNumPy配列に変換
    if isinstance(embeddings, torch.Tensor):
        embeddings = embeddings.detach().cpu().numpy()
    
    # シーケンス内の各トークンペアの類似度を計算
    batch_idx = 0
    seq = embeddings[batch_idx]  # (seq_length, embedding_dim)
    
    print("埋め込み類似度分析:")
    print("=" * 60)
    
    tokens = vocab.decode(token_ids[batch_idx])
    
    for i in range(len(tokens)-1):
        for j in range(i+1, len(tokens)):
            if tokens[i] == '<PAD>' or tokens[j] == '<PAD>':
                continue
            
            # コサイン類似度
            similarity = 1 - cosine(seq[i], seq[j])
            
            print(f"{tokens[i]:10} <-> {tokens[j]:10}: "
                  f"類似度 = {similarity:6.3f}")

# 使用例
analyze_embeddings(embeddings, batch_ids, vocab)

# 出力例:
# 埋め込み類似度分析:
# ============================================================
# 自然        <-> 言語        : 類似度 =  0.021
# 自然        <-> 処理        : 類似度 = -0.045
# 言語        <-> 処理        : 類似度 =  0.089
#
# （: ランダム初期化なので意味のある関係はない
#    学習後は意味的に関連するトークン同士の類似度が高くなる）
```

## 6. ステップ5: シーケンス統合

### 6.1 複数手法

```python
class SequencePooling:
    """複数のシーケンス統合手法"""
    
    @staticmethod
    def mean_pooling(embeddings, attention_mask):
        """
        平均プーリング
        パディングトークンを除いた平均
        
        Args:
            embeddings: (batch_size, seq_length, embedding_dim)
            attention_mask: (batch_size, seq_length)
        """
        # マスクを実行
        mask_expanded = np.expand_dims(attention_mask, axis=-1)
        sum_embedding = np.sum(embeddings * mask_expanded, axis=1)
        sum_mask = np.sum(mask_expanded, axis=1)
        
        # 平均
        pooled = sum_embedding / np.maximum(sum_mask, 1e-9)
        
        return pooled
    
    @staticmethod
    def cls_pooling(embeddings):
        """
        CLS プーリング
        （BERT方式：最初のトークンの埋め込みを使用）
        """
        return embeddings[:, 0, :]
    
    @staticmethod
    def max_pooling(embeddings, attention_mask):
        """
        最大値プーリング
        各次元の最大値を取得
        """
        embeddings_masked = embeddings.copy()
        
        # パディング位置を-infマスク
        mask_expanded = np.expand_dims(attention_mask, axis=-1)
        embeddings_masked = np.where(
            mask_expanded == 1, 
            embeddings_masked, 
            -np.inf
        )
        
        # 最大値を取得
        pooled = np.max(embeddings_masked, axis=1)
        
        return pooled
    
    @staticmethod
    def last_pooling(embeddings, attention_mask):
        """
        最後の有効トークンのプーリング
        """
        batch_size = embeddings.shape[0]
        pooled = np.zeros_like(embeddings[:, 0, :])
        
        for i in range(batch_size):
            last_idx = np.sum(attention_mask[i]) - 1
            pooled[i] = embeddings[i, int(last_idx), :]
        
        return pooled

# 使用例
```

### 7.1 統合パイプライン

```python
class TextToEmbeddingPipeline:
    """テキスト入力から埋め込みベクトルまでの完全パイプライン"""
    
    def __init__(self, vocab, embedding_dim=768, device='cpu'):
        self.vocab = vocab
        self.embedding_layer = EmbeddingLayer(len(vocab), embedding_dim, device)
        self.processor = BatchProcessor(vocab)
        self.pooling = SequencePooling()
        self.device = device
    
    def process(self, texts, max_length=None, pooling_method='mean'):
        """
        テキストのリストを埋め込みベクトルに変換
        
        Args:
            texts: トークンのリストのリスト
            max_length: パディング最大長
            pooling_method: 'mean', 'cls', 'max', 'last'
        
        Returns:
            最終埋め込みベクトル (batch_size, embedding_dim)
        """
        # ステップ1: エンコード + パディング
        padded_ids = self.processor.encode_texts(texts, max_length)
        attention_mask = self.processor.create_attention_mask(padded_ids)
        
        # ステップ2: 埋め込み層
        embeddings = self.embedding_layer.forward(padded_ids)
        embeddings_np = embeddings.detach().cpu().numpy()
        
        # ステップ3: プーリング
        if pooling_method == 'mean':
            pooled = self.pooling.mean_pooling(embeddings_np, attention_mask)
        elif pooling_method == 'cls':
            pooled = self.pooling.cls_pooling(embeddings_np)
        elif pooling_method == 'max':
            pooled = self.pooling.max_pooling(embeddings_np, attention_mask)
        elif pooling_method == 'last':
            pooled = self.pooling.last_pooling(embeddings_np, attention_mask)
        else:
            raise ValueError(f"Unknown pooling method: {pooling_method}")
        
        return pooled
    
    def process_single(self, text_tokens, pooling_method='mean'):
        """単一テキストを処理"""
        return self.process([text_tokens], pooling_method)[0]

# 使用例
```

### 8.1 プロジェクトの実装（src/embeddings/embedding.py）

```python
# === プロジェクトの実装 ===
# モデル: BAAI/bge-m3 (768次元)

def encode_texts(tokenizer, model, texts, batch_size=BATCH_SIZE):
    """
    bge-m3 モデルでテキストをエンコード
    """
    all_embeddings = []

    for i in range(0, len(texts), batch_size):
        batch = texts[i:i+batch_size]

        # ステップ1-2: トークン化（transformers ライブラリ内で実行）
        inputs = tokenizer(
            batch,
            padding=True,
            truncation=True,
            return_tensors="pt",
            max_length=8192,
        ).to(DEVICE)

        # ステップ3-6: 埋め込みとプーリング
        with torch.no_grad():
            outputs = model(**inputs)
            # CLS プーリング（最初のトークン）
            emb = outputs.last_hidden_state[:, 0]
            # L2正規化
            emb = torch.nn.functional.normalize(emb, p=2, dim=1)

        all_embeddings.append(emb.cpu().numpy())

    return np.vstack(all_embeddings)

# === このガイドの実装との関係 ===
# プロジェクト実装                 このガイド
# ├─ tokenizer(...)            ← ステップ2-3（BertTokenizer相当）
# ├─ model(**inputs)           ← ステップ4-6（埋め込み＋プーリング）
# └─ normalize(emb, p=2)       ← 正規化（参考資料で説明）
```
