```python
from src.self_improvement.context_analyzer import ContextAnalyzer, ImplicitIntentDetector, MetaContextTracker
from src.self_improvement.domain_knowledge import DomainKnowledgeManager

class Phase7QueryPreprocessor:
    """Phase 7対応クエリプリプロセッサ"""
    
    def __init__(self):
        self.context_analyzer = ContextAnalyzer()
        self.intent_detector = ImplicitIntentDetector()
        self.meta_tracker = MetaContextTracker()
        self.domain_manager = DomainKnowledgeManager()
    
    def preprocess(self, query: str, user_id: str = None) -> dict:
        """
        クエリの事前処理
        
        Returns:
            - query_context: 文脈分析結果
            - implicit_intents: 隠れた意図
            - primary_domain: 主要ドメイン
            - related_domains: 関連ドメイン
            - user_profile: ユーザープロフィール
        """
        # 文脈分析
        query_context = self.context_analyzer.analyze_query(query)
        
        # 隠れた意図検出
        implicit_intents = self.intent_detector.detect(query)
        
        # ドメイン推定
        primary_domain = self.domain_manager.infer_domain_from_query(query)
        related_domains = self.domain_manager.infer_related_domains(primary_domain)
        
        # ユーザー知識追跡（オプション）
        user_profile = None
        if user_id:
            user_profile = self.meta_tracker.get_user_profile(user_id)
        
        return {
            'query_context': query_context,
            'implicit_intents': implicit_intents,
            'primary_domain': primary_domain,
            'related_domains': related_domains,
            'user_profile': user_profile
        }
```

#### 2.2: マルチドメイン知識検索の追加

**ファイル**: `src/rag/multi_domain_retriever.py` (新規作成)

```python
from src.self_improvement.domain_knowledge import DomainKnowledgeManager, CrossDomainLinker, DomainIndexer
from src.self_improvement.reasoning_engine import KnowledgeIntegrator

class MultiDomainRetriever:
    """マルチドメイン対応リトリーバー"""
    
    def __init__(self, embedding_model, corpus_index):
        self.domain_manager = DomainKnowledgeManager()
        self.cross_linker = CrossDomainLinker()
        self.domain_indexer = DomainIndexer()
        self.knowledge_integrator = KnowledgeIntegrator()
        
        self.embedding_model = embedding_model
        self.corpus_index = corpus_index
    
    def retrieve_multi_domain(self, query: str, primary_domain: str, 
                             related_domains: List[str], top_k: int = 5) -> dict:
        """
        マルチドメイン知識の取得と統合
        
        Returns:
            - primary_results: 主要ドメイン結果
            - cross_domain_results: クロスドメイン結果
            - integrated: 統合結果
        """
        # 主要ドメイン検索
        primary_results = self._search_domain(query, primary_domain, top_k)
        
        # 関連ドメイン検索
        cross_domain_results = {}
        for related_domain in related_domains:
            cross_domain_results[related_domain] = self._search_domain(
                query, related_domain, top_k // len(related_domains)
            )
        
        # 知識統合
        integrated = self.knowledge_integrator.integrate_knowledge([
            {'domain': primary_domain, 'facts': primary_results},
            {'domain': d, 'facts': cross_domain_results[d]} 
            for d in related_domains
        ])
        
        return {
            'primary_results': primary_results,
            'cross_domain_results': cross_domain_results,
            'integrated': integrated
        }
    
    def _search_domain(self, query: str, domain: str, top_k: int) -> List[Document]:
        """特定ドメイン内での検索"""
        # ドメイン固有のキーワード抽出
        domain_keywords = self.domain_manager.get_domain_keywords(domain)
        
        # ドメイン重み付きで検索
        embeddings = self.embedding_model.encode([query] + list(domain_keywords))
        results = self.corpus_index.search(embeddings[0], top_k)
        
        return results
```

```python
                    causality, uncertainty, user_profile):

    prompt = f"""
以下のコンテキストでユーザーの質問に回答してください。

{query}

【ユーザーの隠れた意図】
{', '.join(implicit_intents.keys())}

【関連知識】
{_format_context(context_documents)}

【クロスドメイン見解】
{_format_causality(causality)}

【不確実性レベル】
{uncertainty.level} ({uncertainty.sources})

【ユーザー知識レベル】
{user_profile.get('knowledge_level', 'general') if user_profile else 'general'}

        return response + uncertainty_note
    
    return response
```

```python
# src/rag/multi_domain_retriever.py に追加

from functools import lru_cache
import hashlib

class OptimizedMultiDomainRetriever(MultiDomainRetriever):
    """パフォーマンス最適化版"""
    
    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self._domain_cache = {}
        self._query_cache = {}
    
    @lru_cache(maxsize=256)
    def _get_domain_keywords_cached(self, domain: str):
        """ドメインキーワードのキャッシング"""
        return self.domain_manager.get_domain_keywords(domain)
    
    def retrieve_multi_domain_cached(self, query: str, primary_domain: str, 
                                    related_domains: List[str], top_k: int = 5) -> dict:
        """キャッシュを活用した取得"""
        
        # クエリハッシュ作成
        query_hash = hashlib.md5(
            f"{query}_{primary_domain}_{','.join(related_domains)}_{top_k}".encode()
        ).hexdigest()
        
        # キャッシュ確認
        if query_hash in self._query_cache:
            return self._query_cache[query_hash]
        
        # 検索実行
        results = self.retrieve_multi_domain(query, primary_domain, related_domains, top_k)
        
        # キャッシュ保存
        self._query_cache[query_hash] = results
        
        return results
```

#### 4.2: 並列処理

```python
# src/rag/multi_domain_retriever.py に追加

import asyncio
from concurrent.futures import ThreadPoolExecutor

class AsyncMultiDomainRetriever(MultiDomainRetriever):
    """非同期処理版"""
    
    async def retrieve_multi_domain_async(self, query: str, primary_domain: str, 
                                         related_domains: List[str], top_k: int = 5) -> dict:
        """非同期マルチドメイン検索"""
        
        # 並列に複数ドメインを検索
        tasks = [
            asyncio.create_task(self._search_domain_async(query, primary_domain, top_k))
        ]
        tasks.extend([
            asyncio.create_task(self._search_domain_async(query, domain, top_k // len(related_domains)))
            for domain in related_domains
        ])
        
        results = await asyncio.gather(*tasks)
        
        return {
            'primary_results': results[0],
            'cross_domain_results': {
                domain: result for domain, result in zip(related_domains, results[1:])
            },
            'integrated': await self._integrate_async(results)
        }
    
    async def _search_domain_async(self, query: str, domain: str, top_k: int):
        """非同期ドメイン検索"""
        loop = asyncio.get_event_loop()
        return await loop.run_in_executor(
            None, self._search_domain, query, domain, top_k
        )
```
