# validators/common.py
from functools import wraps
from typing import Any, Callable

class ValidationError(Exception):
    """バリデーション エラー基底クラス"""
    pass

def validate_email(email: str) -> bool:
    """メールアドレスバリデーション"""
    if not isinstance(email, str):
        raise TypeError("Email must be string")
    if len(email) < 5:
        raise ValidationError("Email too short")
    if '@' not in email:
        raise ValidationError("Invalid email format")
    return True

def validate_input(validator_func: Callable) -> Callable:
    """バリデーションデコレータ"""
    @wraps(validator_func)
    def wrapper(*args, **kwargs):
        try:
            return validator_func(*args, **kwargs)
        except (TypeError, ValidationError) as e:
            raise ValidationError(f"{validator_func.__name__} failed: {str(e)}")
    return wrapper

# auth.py
from validators.common import validate_email

@validate_input
def authenticate_user(email):
    validate_email(email)
    # ... rest of code

# user.py
from validators.common import validate_email

@validate_input
def create_user(email):
    validate_email(email)
    # ... rest of code

# compliance.py
from validators.common import validate_email

@validate_input
def log_audit_event(email):
    validate_email(email)
    # ... rest of code

from typing import List, Dict, Callable, Any
from dataclasses import dataclass
from datetime import datetime

@dataclass
class Event:
    id: str
    timestamp: datetime
    data: Dict[str, Any]
    severity: int

@dataclass
class ProcessedEvent:
    id: str
    timestamp: datetime
    data: Dict[str, Any]

def process_events(
    events: List[Event],
    filter_func: Callable[[Event], bool]
) -> List[ProcessedEvent]:
    """イベント処理"""
    results: List[ProcessedEvent] = []
    
    for event in events:
        if filter_func(event):
            processed = ProcessedEvent(
                id=event.id,
                timestamp=event.timestamp,
                data=event.data
            )
            results.append(processed)
    
    return results

    events,

### 4. 非同期処理最適化

**Before: ブロッキング処理**

```python
import requests
import time

def fetch_multiple_threats(threat_ids: List[str]) -> List[Dict]:
    """複数の脅威情報を取得 (遅い)"""
    results = []
    for threat_id in threat_ids:
        response = requests.get(f"https://api/threats/{threat_id}")
        results.append(response.json())
        time.sleep(0.1)  # 部分的な遅延対策
    
    return results

# 実行時間: 10 脅威 × 1 秒 = 10 秒
```

**After: 非同期処理**

```python
import asyncio
import aiohttp
from typing import List, Dict

async def fetch_threat(
    session: aiohttp.ClientSession,
    threat_id: str
) -> Dict:
    """脅威情報を非同期で取得"""
    async with session.get(f"https://api/threats/{threat_id}") as response:
        return await response.json()

async def fetch_multiple_threats(threat_ids: List[str]) -> List[Dict]:
    """複数の脅威情報を並列取得 (高速)"""
    async with aiohttp.ClientSession() as session:
        tasks = [
            fetch_threat(session, threat_id)
            for threat_id in threat_ids
        ]
        return await asyncio.gather(*tasks)

# 実行時間: 最大 1 秒 (すべて並列)
# 高速化: 10倍

# メイン実行
if __name__ == "__main__":
    threat_ids = [f"threat_{i}" for i in range(10)]
    results = asyncio.run(fetch_multiple_threats(threat_ids))
```

**期待効果:** スループット +10x (1000 req/s → 10,000 req/s)

```python
import logging
from pythonjsonlogger import jsonlogger
from logging.handlers import RotatingFileHandler

# config/logging_config.py
def setup_logging(service_name: str):
    """ロギング統一設定"""
    
    # ルートロガー
    root_logger = logging.getLogger()
    root_logger.setLevel(logging.DEBUG)
    
    # JSON フォーマッター (ELK/Datadog 対応)
    json_formatter = jsonlogger.JsonFormatter(
        '%(timestamp)s %(level)s %(name)s %(message)s %(service)s'
    )
    
    # ファイルハンドラ (ローテーション)
    file_handler = RotatingFileHandler(
        f'logs/{service_name}.log',
        maxBytes=100_000_000,  # 100MB
        backupCount=10
    )
    file_handler.setFormatter(json_formatter)
    root_logger.addHandler(file_handler)
    
    # コンソールハンドラ
    console_handler = logging.StreamHandler()
    console_handler.setFormatter(json_formatter)
    root_logger.addHandler(console_handler)
    
    return root_logger

# main.py
setup_logging('esp-authentication')

# auth.py
logger = logging.getLogger('esp-authentication.auth')

def authenticate(user: str) -> bool:
    logger.info("Authentication attempt", extra={'user': user})
    # ... auth logic
    logger.info("Authentication successful", extra={'user': user})
    return True

# threat_detection.py
logger = logging.getLogger('esp-authentication.threat')

def detect_threat(event: Dict) -> bool:
    logger.warning(
        "Threat detected",
        extra={
            'event_id': event['id'],
            'severity': event['severity'],
            'timestamp': event['timestamp']
        }
    )
    return True
```
